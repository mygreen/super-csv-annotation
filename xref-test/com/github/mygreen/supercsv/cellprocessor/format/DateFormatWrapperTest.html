126     @Test
127     public void testParseAndPrintWithMultiThread() {
128         
129         Date baseDate = toDate(2000, 1, 1);
130         
131         List<Date> list = IntStream.range(0, 99)
132                 .mapToObj(i -> plusDays(baseDate, i))
133                 .collect(Collectors.toList());
134         
135         // with parallelStream
136         list.parallelStream().forEach(d -> {
137             try {
138                 String str = formatter.print(d);
139                 Date date = formatter.parse(str);
140                 assertThat(date).isEqualTo(d);
141                 
142             } catch (Exception e) {
143                 e.printStackTrace();
144                 fail();
145             }
146         });
147         
148         // with executor
149         ExecutorService executor = Executors.newFixedThreadPool(5);
150         try {
151             final CountDownLatch countDown = new CountDownLatch(list.size());
152             list.stream().forEach(d -> {
153                 
154                 executor.submit(() -> {
155                     try {
156                         String str = formatter.print(d);
157                         Date date = formatter.parse(str);
158                         assertThat(date).isEqualTo(d);
159                         
160                     } catch (Exception e) {
161                         fail();
162                     } finally {
163                         countDown.countDown();
164                     }
165                 });
166                 
167             });
168             
169             countDown.await();
170             
171         } catch (Exception e) {
172             e.printStackTrace();
173             fail();
174             
175         } finally {
176             executor.shutdown();
177         }
178         
179     }
