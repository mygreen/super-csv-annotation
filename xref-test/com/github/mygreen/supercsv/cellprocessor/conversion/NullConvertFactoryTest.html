67      @Test
68      public void testCreate_string() {
69          
70          FieldAccessor field = getFieldAccessor(TestCsv.class, "col_string", comparator);
71          StringProcessorBuilder builder = (StringProcessorBuilder) builderResolver.resolve(String.class);
72          TextFormatter<String> formatter = builder.getFormatter(field, config);
73          
74          CsvNullConvert anno = field.getAnnotationsByGroup(CsvNullConvert.class, groupEmpty).get(0);
75          {
76              //next null
77              Optional<CellProcessor> processor = factory.create(anno, Optional.empty(), field, formatter, config);
78              printCellProcessorChain(processor.get(), name.getMethodName());
79              
80              assertThat(processor.get()).isInstanceOf(NullConvert.class);
81              
82              NullConvert actual = (NullConvert)processor.get();
83              {
84                  // input not match 
85                  String input = "abcde";
86                  assertThat((Object)actual.execute(input, ANONYMOUS_CSVCONTEXT)).isEqualTo(input);
87                  
88              }
89              
90              {
91                  // input null 
92                  String input = null;
93                  assertThat((Object)actual.execute(input, ANONYMOUS_CSVCONTEXT)).isEqualTo(input);
94                  
95              }
96              
97              {
98                  // input match
99                  String input = "-";
100                 assertThat((Object)actual.execute(input, ANONYMOUS_CSVCONTEXT)).isNull();
101                 
102             }
103             
104         }
105         
106         {
107             //next exist
108             Optional<CellProcessor> processor = factory.create(anno, Optional.of(new NextCellProcessor()), field, formatter, config);
109             printCellProcessorChain(processor.get(), name.getMethodName());
110             
111             assertThat(processor.get()).isInstanceOf(NullConvert.class);
112             
113             NullConvert actual = (NullConvert)processor.get();
114             {
115                 // input not match
