84      @CsvBean
85      private static class TestCsv {
86          
87          @CsvColumn(number=1)
88          @NotEmpty
89          private String id;
90          
91          @CsvColumn(number=2)
92          @Length(max=10)
93          @Pattern(regexp="[\\p{Alnum}]+", message="半角英数字で設定してください。", groups=Group1.class)
94          private String name;
95          
96          @CsvColumn(number=3)
97          @Range(min=0, max=100, groups=Group2.class)
98          private Integer age;
99          
100         @CsvColumn(number=4)
101         boolean used;
102         
103         @AssertTrue(message="{name}が設定されている際には、{age}は必須です。", groups=Group2.class)
104         boolean isValidAgeRequired() {
105             if(name != null && !name.isEmpty()) {
106                 return age != null;
107             }
108             
109             return false;
110         }
111         
112         @DecimalMax(value="20", groups=Group3.class)
113         Integer getAge() {
114             return age;
115         }
116         
117         @AssertTrue(groups=Group4.class)
118         boolean isUsed() {
119             return used;
120         }
