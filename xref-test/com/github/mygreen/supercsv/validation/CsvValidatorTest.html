106     public static class TestValidator implements CsvValidator<TestCsv> {
107         
108         @Override
109         public void validate(final TestCsv record, final CsvBindingErrors bindingErrors,
110                 final ValidationContext<TestCsv> validationContext) {
111             
112             final CsvField<Integer> ageField = new CsvField<>(validationContext, record, "age");
113             
114             final CsvField<Integer> salaryField = new CsvField<>(validationContext, record, "salary");
115             salaryField
116                 .add(new CsvFieldValidator<Integer>() {
117                     
118                     @Override
119                     public void validate(final CsvBindingErrors bindingErrors, final CsvField<Integer> field) {
120                         if(ageField.isEmpty()) {
121                             return;
122                         }
123                         
124                         // カラム「age(年齢)」が20以上の場合、カラム「給料(salary)」が設定されているかチェックする。
125                         if(ageField.isNotEmpty() && ageField.getValue() >= 20 && field.isEmpty()) {
126                             
127                             // メッセージ中の変数の作成
128                             final Map<String, Object> vars = createMessageVariables(field);
129                             vars.put("maxAge", 20);
130                             
131                             // ageに関するフィールドエラーの追加
132                             bindingErrors.rejectValue(field.getName(), field.getType(), "age.required", vars);
133                         }
134                     }
135                 })
136                 .add(new MaxValidator(10_000_000))
137                 .validate(bindingErrors);
138             
139             
140         }
141         
142     }

144     public static class MaxValidator implements CsvFieldValidator<Integer> {
145         
146         private final int max;
147         
148         public MaxValidator(final int max) {
149             this.max = max;
150         }
151         
152         @Override
153         public void validate(CsvBindingErrors bindingErrors, CsvField<Integer> field) {
154             if(field.isEmpty()) {
155                 return;
156             }
157             
158             if(field.getValue() > max) {
159                 // メッセージ変数の組み立て
160                 Map<String, Object> vars = createMessageVariables(field);
161                 vars.put("max", max);
162                 
163                 bindingErrors.rejectValue(field.getName(), field.getType(), "fieldError.max", vars);
164             }
165         }

169     @Test
170     public void testValidate_read() throws Exception {
171         
172         String csv = "名前,年齢,給料"
173                 + "\r\n阿部真一, 10, "
174                 + "\r\n山田太郎, 20, \"200,000\""
175                 + "\r\n鈴木次郎, 30, "  // 給料がnull
176                 + "\r\n林花子, 40, \"20,000,000\"" // 給料が上限値超え
177                 + "\r\n本田一郎, 50, abc" // 給料が数値
178                 ;
179         
180         CsvAnnotationBeanReader<TestCsv> csvReader = new CsvAnnotationBeanReader<>(TestCsv.class, new StringReader(csv), CsvPreference.STANDARD_PREFERENCE);
181         exceptionConverter.setMessageResolver(testMessageResolver);
182         csvReader.setExceptionConverter(exceptionConverter);
183         
184         List<TestCsv> list = csvReader.readAll(true);
185         csvReader.close();
186         
187         assertThat(list).hasSize(2);
188         assertThat(list.get(0)).hasFieldOrPropertyWithValue("name", "阿部真一");
189         assertThat(list.get(1)).hasFieldOrPropertyWithValue("name", "山田太郎");
190         
191         List<String> errorMessages = csvReader.getErrorMessages();
192         csvReader.getErrorMessages().stream().forEach(System.out::println);
193         
194         assertThat(errorMessages).hasSize(3).containsExactly(
195                 "[4行, 3列] : 項目「給料」は、年齢が20歳以上の場合には必須です。",
196                 "[5行, 3列] : 項目「給料」の値（20,000,000）は、10,000,000以内で入力してください。",
197                 "[6行, 3列] : 項目「給料」の値（abc）は、整数の書式「#,##0」として不正です。"
198                 );
199         
200     }

203      * 書き込み時の入力値検証 - 検証のスキップ

205     @Test
206     public void testValidate_write_skipValidate() throws Exception {
207         
208         StringWriter writer = new StringWriter();
209         
210         beanMappingFactory.getConfiguration().setSkipValidationOnWrite(true);
211         BeanMapping<TestCsv> beanMapping = beanMappingFactory.create(TestCsv.class);
212         
213         CsvAnnotationBeanWriter<TestCsv> csvWriter = new CsvAnnotationBeanWriter<>(beanMapping, writer, CsvPreference.STANDARD_PREFERENCE);
214         exceptionConverter.setMessageResolver(testMessageResolver);
215         csvWriter.setExceptionConverter(exceptionConverter);
216         
217         List<TestCsv> list = new ArrayList<>();
218         list.add(new TestCsv("阿部真一", 10, null));
219         list.add(new TestCsv("山田太郎", 20, 200_000));
220         list.add(new TestCsv("鈴木次郎", 30, null));  // 給料がnull
221         list.add(new TestCsv("林花子", 40, 20_000_000));
222         list.add(new TestCsv("橋本誠", 60, 400_000)); // 給料が上限値超え
223         
224         csvWriter.writeAll(list, true);
225         csvWriter.flush();
226         csvWriter.close();
227         
228         
229         List<String> errorMessages = csvWriter.getErrorMessages();
230         assertThat(errorMessages).isEmpty();
231         
232         String csv = writer.toString();
233         System.out.println(csv);
234         
235         String expected = "名前,年齢,給料"
236                 + "\r\n阿部真一,10,"
237                 + "\r\n山田太郎,20,\"200,000\""
238                 + "\r\n鈴木次郎,30,"  // 給料がnull
239                 + "\r\n林花子,40,\"20,000,000\"" // 給料が上限値超え
240                 + "\r\n橋本誠,60,\"400,000\""
241                 + "\r\n"
242                 ;
243         
244         assertThat(csv).isEqualTo(expected);
245         
246
