25  public class CustomFunctionsTest {
26      
27      private TextPrinter<Integer> printer;
28      
29      @Before
30      public void setUp() throws Exception {
31          this.printer = (value) -> new DecimalFormat("#,##0").format(value);
32          
33      }
34      
35      @Test
36      public void testDefaultString() {
37          
38          assertThat(CustomFunctions.defaultString(null)).isEqualTo("");
39          assertThat(CustomFunctions.defaultString("")).isEqualTo("");
40          assertThat(CustomFunctions.defaultString("abc")).isEqualTo("abc");
41          
42      }
43      
44      /**
45       * {@link CustomFunctions#join(int[], String)}
46       */
47      @Test
48      public void testJoin_int_array() {
49          
50          assertThat(CustomFunctions.join((int[])null, ", ")).isEqualTo("");
51          assertThat(CustomFunctions.join(new int[]{}, ", ")).isEqualTo("");
52          assertThat(CustomFunctions.join(new int[]{1,2,3}, ", ")).isEqualTo("1, 2, 3");
53          assertThat(CustomFunctions.join(new int[]{1,2,3}, null)).isEqualTo("123");
54          
55      }
56      
57      /**
58       * {@link CustomFunctions#join(Object[], String)}
59       */
60      @Test
61      public void testJoin_object_array() {
62          
63          assertThat(CustomFunctions.join((Object[])null, ", ")).isEqualTo("");
64          assertThat(CustomFunctions.join(new Object[]{}, ", ")).isEqualTo("");
65          assertThat(CustomFunctions.join(new Object[]{1,2,3}, ", ")).isEqualTo("1, 2, 3");
66          assertThat(CustomFunctions.join(new Object[]{1,2,3}, null)).isEqualTo("123");
67          
68      }
69      
70      /**
71       * {@link CustomFunctions#join(Object[], String, TextPrinter)}
72       */
73      @Test
74      public void testJoin_object_array_printer() {
75          
76          Object[] input = new Object[]{1000, 2000, 3000};
77          
78          assertThatThrownBy(() -> CustomFunctions.join((Object[])null, ", ", null)).isInstanceOf(NullPointerException.class);
79          assertThat(CustomFunctions.join((Object[])null, ", ", printer)).isEqualTo("");
80          assertThat(CustomFunctions.join(new Object[]{}, ", ", printer)).isEqualTo("");
81          assertThat(CustomFunctions.join(input, ", ", printer)).isEqualTo("1,000, 2,000, 3,000");
82          assertThat(CustomFunctions.join(input, null, printer)).isEqualTo("1,0002,0003,000");
83          
84      }
85      
86      /**
87       * {@link CustomFunctions#join(java.util.Collection, String)}
88       */
89      @Test
90      public void testJoin_collection() {
91          
92          Collection<Integer> input = Arrays.asList(1000, 2000, 3000);
93          
94          assertThat(CustomFunctions.join((Collection<Integer>)null, ", ")).isEqualTo("");
95          assertThat(CustomFunctions.join(Collections.emptyList(), ", ")).isEqualTo("");
96          assertThat(CustomFunctions.join(input, ", ")).isEqualTo("1000, 2000, 3000");
97          assertThat(CustomFunctions.join(input, null)).isEqualTo("100020003000");
98          
99      }
