334     public static void printCellProcessorChain(final CellProcessor cellProcessor, final PrintWriter writer) {
335         
336         String index = "";
337         CellProcessor cp = cellProcessor;
338         do {
339             if(index.length() == 0) {
340                 writer.printf("%s%s\n", index, cp.getClass().getName());
341                 writer.flush();
342             } else {
343                 writer.printf("%s└%s\n", index, cp.getClass().getName());
344                 writer.flush();
345             }
346             
347             // next processor
348             try {
349                 if(cp instanceof CellProcessorAdaptor) {
350                     Field field = CellProcessorAdaptor.class.getDeclaredField("next");
351                     field.setAccessible(true);
352                     cp = (CellProcessor) field.get(cp);
353                 } else {
354                     break;
355                 }
356                 
357             } catch(ReflectiveOperationException e) {
358                 return;
359             }
360             
361             index += "    ";
362             
363         } while(cp != null);
364         
365     }

396      * ランダムに、大文字、小文字に変換する。
397      * @param str
398      * @return
399      */
400     public static final String toRandomCase(final String str) {
401         
402         StringBuilder sb = new StringBuilder();
403         try {
404             SecureRandom random = new SecureRandom();
405             final int len = str.length();
406             for(int i=0; i < len; i++) {
407                 char c = str.charAt(i);
408                 if(random.nextBoolean()) {
409                     sb.append(String.valueOf(c).toLowerCase());
410                 } else {
411                     sb.append(String.valueOf(c).toUpperCase());
412                 }
413             }
414         } catch(Exception e) {
415             throw new RuntimeException(e);
416         }
417         return sb.toString();
418         
419     }
