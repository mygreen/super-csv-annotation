22      /**
23       * カラム番号が重複しているかチェックする。また、番号が1以上かもチェ句する。
24       * @param beanType Beanタイプ
25       * @param list カラム情報の一覧
26       * @return チェック済みの番号
27       */
28      public static TreeSet<Integer> validateDuplicatedColumnNumber(final Class<?> beanType, final List<ColumnMapping> list) {
29          
30          final TreeSet<Integer> checkedNumber = new TreeSet<>();
31          final TreeSet<Integer> duplicatedNumbers = new TreeSet<>();
32          for(ColumnMapping columnMapping : list) {
33              
34              if(checkedNumber.contains(columnMapping.getNumber())) {
35                  duplicatedNumbers.add(columnMapping.getNumber());
36              }
37              checkedNumber.add(columnMapping.getNumber());
38              
39          }
40          
41          if(!duplicatedNumbers.isEmpty()) {
42              // 重複している 属性 numberが存在する場合
43              throw new SuperCsvInvalidAnnotationException(MessageBuilder.create("anno.attr.duplicated")
44                      .var("property", beanType.getName())
45                      .varWithAnno("anno", CsvColumn.class)
46                      .var("attrName", "number")
47                      .var("attrValues", duplicatedNumbers)
48                      .format());
49          }
50          
51          // カラム番号が1以上かチェックする
52          final int minColumnNumber = checkedNumber.first();
53          if(minColumnNumber <= 0) {
54              throw new SuperCsvInvalidAnnotationException(MessageBuilder.create("anno.attr.min")
55                      .var("property", beanType.getName())
56                      .varWithAnno("anno", CsvColumn.class)
57                      .var("attrName", "number")
58                      .var("attrValue", minColumnNumber)
59                      .var("min", 1)
60                      .format());
61              
62          }
63          
64          return checkedNumber;
65      }

67      /**
68       * 欠けているカラム番号がある場合、その番号を持つダミーのカラムを追加する。
69       * @param beanType Beanタイプ
70       * @param list カラム情報の一覧
71       * @param partialAnno Beanに設定されているアノテーション{@link CsvPartial}の情報。
72       * @param suppliedHeaders 提供されたヘッダー。提供されてない場合は、長さ0の配列。
73       * @return
74       */
75      public static TreeSet<Integer> supplyLackedNumberMappingColumn(final Class<?> beanType, final List<ColumnMapping> list,
76              final Optional<CsvPartial> partialAnno, final String[] suppliedHeaders) {
77          
78          final TreeSet<Integer> checkedNumber = list.stream()
79                  .filter(col -> col.isDeterminedNumber())
80                  .map(col -> col.getNumber())
81                  .collect(Collectors.toCollection(TreeSet::new));
82          
83          // 定義されている列番号の最大値
84          final int maxColumnNumber = checkedNumber.last();
85          
86          // Beanに定義されていない欠けているカラム番号の取得
87          final TreeSet<Integer> lackedNumbers = new TreeSet<Integer>();
88          for(int i=1; i <= maxColumnNumber; i++) {
89              if(!checkedNumber.contains(i)) {
90                  lackedNumbers.add(i);
91              }
92          }
93          
94          // 定義されているカラム番号より、大きなカラム番号を持つカラム情報の補足
95          if(partialAnno.isPresent()) {
96              
97              final int partialColumnSize = partialAnno.get().columnSize();
98              if(maxColumnNumber > partialColumnSize) {
99                  throw new SuperCsvInvalidAnnotationException(partialAnno.get(), MessageBuilder.create("anno.CsvPartial.columSizeMin")
100                         .var("property", beanType.getName())
101                         .var("columnSize", partialColumnSize)
102                         .var("maxColumnNumber", maxColumnNumber)
103                         .format());
104                 
105             }
106             
107             if(maxColumnNumber < partialColumnSize) {
108                 for(int i= maxColumnNumber+1; i <= partialColumnSize; i++) {
109                     lackedNumbers.add(i);
110                 }
111             }
112             
113         }
114         
115         // 不足分のカラムがある場合は、部分的な読み書き用カラムとして追加する
116         if(lackedNumbers.size() > 0) {
117             
118             for(int number : lackedNumbers) {
119                 list.add(createPartialColumnMapping(number, partialAnno, getSuppliedHeaders(suppliedHeaders, number)));
120             }
121             
122             list.sort(null);
123         }
124         
125         return lackedNumbers;
126         
127     }

129     /**
130      * 提供されたヘッダーから該当するカラム番号のヘッダーを取得する。
131      * @param suppliedHeaders 提供されたヘッダー。提供されてない場合は、長さ0の配列。
132      * @param columnNumber カラム番号。1から始まる。
133      * @return 該当するカラムのヘッダー。見つからない場合は空を返す。
134      */
135     private static Optional<String> getSuppliedHeaders(final String[] suppliedHeaders, final int columnNumber) {
136         
137         final int length = suppliedHeaders.length;
138         if(length == 0) {
139             return Optional.empty();
140         }
141         
142         if(columnNumber < length) {
143             return Optional.ofNullable(suppliedHeaders[columnNumber-1]);
144         }
145         
146         return Optional.empty();
147         
148     }

150     /**
151      * 部分的なカラムの場合の作成
152      * @param columnNumber 列番号
153      * @param partialAnno アノテーション {@literal @CsvPartial}のインスタンス
154      * @param suppliedHeader 補完対象のヘッダーの値
155      * @return 部分的なカラム情報。
156      */
157     private static ColumnMapping createPartialColumnMapping(int columnNumber, final Optional<CsvPartial> partialAnno,
158             final Optional<String> suppliedHeader) {
159         
160         final ColumnMapping columnMapping = new ColumnMapping();
161         columnMapping.setNumber(columnNumber);
162         columnMapping.setPartialized(true);
163         
164         String label = String.format("column%d", columnNumber);
165         
166         if(suppliedHeader.isPresent()) {
167             label = suppliedHeader.get();
168         }
169         
170         if(partialAnno.isPresent()) {
171             for(CsvPartial.Header header : partialAnno.get().headers()) {
172                 if(header.number() == columnNumber) {
173                     label = header.label();
174                     break;
175                 }
176             }
177         }
178         columnMapping.setLabel(label);
179         
180         return columnMapping;
181         
182     }
