18      private int[] toCodePointArray(final String text) {
19          final int length = text.length();
20          final int codePointLength = text.codePointCount(0, length);
21  
22          int[] array = new int[codePointLength];
23          for(int i=0, j=0, codePoint=0; i < length; j++, i+=Character.charCount(codePoint)) {
24              codePoint = text.codePointAt(i);
25              array[j] = codePoint;
26          }
27  
28          return array;
29      }

31      @Override
32      public String pad(final String text, final int size, final char padChar, final boolean rightAlign, final boolean chopped) {
33  
34          final int currentSize = count(text);
35          final int padCharSize = count(String.valueOf(padChar));
36          final int[] codePointArray = toCodePointArray(text);
37          final int codePointSize = codePointArray.length;
38  
39          if(rightAlign) {
40              // 右詰
41              if(currentSize == size) {
42                  return text;
43  
44              } else if(currentSize > size) {
45                  // 指定した長さを超える場合
46                  if(chopped) {
47                      // 切り出す場合
48  
49                      /*
50                       * 左端から長さをカウントしていき、オーバした長さ分を切り取る
51                       */
52                      final int overLength = currentSize - size;
53                      for(int i=0, chopLength=0; i < codePointSize; i++) {
54                          final int codePoint = codePointArray[i];
55                          chopLength += count(codePoint);
56                          if(chopLength >= overLength) {
57                              // substring(i+1)
58                              String chopText = new String(codePointArray, i+1, codePointSize-(i+1));
59  
60                              // 切り取った後の再調整。
61                              // パディング文字が全角の場合は、余分に切り取る場合があるため。
62                              return pad(chopText, size, padChar, rightAlign, chopped);
63                          }
64  
65                      }
66  
67                      // 全て切り出す場合
68                      return "";
69  
70                  } else {
71                      // 切り出さない場合
72                      return text;
73                  }
74              } else {
75                  // 指定したサイズより少ない場合
76  
77                  /*
78                   * パディング文字を付与していく
79                   * ・パディング文字が全角の時があり、長さが2以上になるので注意する
80                   */
81                  final int lackLength = size - currentSize;
82                  int padCount = lackLength / padCharSize;
83  
84                  StringBuilder appender = new StringBuilder();
85                  for(int i=0; i < padCount; i++) {
86                      appender.append(padChar);
87                  }
88  
89                  appender.append(text);
90  
91                  return appender.toString();
92  
93              }
94          } else {
95              // 左詰
96  
97              if(currentSize == size) {
98                  return text;
99  
100             } else if(currentSize > size) {
101                 // 指定したサイズを超える場合
102                 if(chopped) {
103                     // 切り出す場合
104 
105                     /*
106                      * 右端から長さをカウントしていき、オーバした長さ分を切り取る
107                      */
108                     final int overLength = currentSize - size;
109                     for(int i=codePointSize-1, chopLength=0; i >= 0; i--) {
110                         final int codePoint = codePointArray[i];
111                         chopLength += count(codePoint);
112                         if(chopLength >= overLength) {
113                             // substring(0, i)
114                             String chopText = new String(codePointArray, 0, i);
115

117                             // パディング文字が全角の場合は、余分に切り取る場合があるため。
118                             return pad(chopText, size, padChar, rightAlign, chopped);
119                         }

125                     return "";
126 
127                 } else {
128                     // 切り出さない場合
129                     return text;
130                 }
131             } else {
132                 // 指定したサイズより少ない場合
133 
134                 /*
135                  * パディング文字を付与していく
136                  * ・パディング文字が全角の時があり、長さが2以上になるので注意する
137                  */
138                 final int lackLength = size - currentSize;
139                 int padCount = lackLength / padCharSize;
