16  /**
17   * アノテーション{@link CsvMultiPad}を元に、CellProcesssor{@link MultiPad}を作成します。
18   *
19   * @since 2.1
20   * @author T.TSUCHIE
21   *
22   */
23  public class MultiPadFactory implements ConversionProcessorFactory<CsvMultiPad> {
24      
25      @Override
26      public Optional<CellProcessor> create(final CsvMultiPad anno, final Optional<CellProcessor> next,
27              final FieldAccessor field, final TextFormatter<?> formatter, final Configuration config) {
28          
29          if(anno.size() <= 0) {
30              throw new SuperCsvInvalidAnnotationException(anno,
31                      MessageBuilder.create("anno.attr.min")
32                          .var("property", field.getNameWithClass())
33                          .varWithAnno("anno", anno.annotationType())
34                          .var("attrName", "size")
35                          .var("attrValue", anno.size())
36                          .var("min", 1)
37                          .format());
38          }
39          
40          final PaddingProcessor paddingProcessor = (PaddingProcessor)config.getBeanFactory().create(anno.paddingProcessor());
41          
42          final MultiPad processor = next.map(n -> new MultiPad(anno.size(), anno.padChar(), anno.rightAlign(),
43                      anno.chopped(), paddingProcessor, (StringCellProcessor)n))
44                  .orElseGet(() -> new MultiPad(anno.size(), anno.padChar(), anno.rightAlign(),
45                          anno.chopped(), paddingProcessor));
46          
47          return Optional.of(processor);
48      }
