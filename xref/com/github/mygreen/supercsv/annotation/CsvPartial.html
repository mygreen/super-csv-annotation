3   import java.lang.annotation.Documented;
4   import java.lang.annotation.ElementType;
5   import java.lang.annotation.Retention;
6   import java.lang.annotation.RetentionPolicy;
7   import java.lang.annotation.Target;
8   
9   import com.github.mygreen.supercsv.annotation.conversion.CsvFixedSize;

11  /**
12   * 部分的にカラムをマッピングする際のカラム情報を補完するためのアノテーションです。
13   * 
14   * <h3 class="description">基本的な使い方</h3>
15   * 
16   * <ul>
17   *   <li>部分的にマッピングする際には、単純に、マッピングしないカラムの定義を省略します。</li>
18   *   <li>ただし、定義している最大のカラム番号が、実ファイルのカラム番号よりも小さい場合は、
19   *       属性{@link #columnSize()}で、実際のカラムサイズを指定します。
20   *   </li>
21   *   <li>見出し行がある場合、定義していない見出しを属性{@link #headers()}で定義します。</li>
22   * </ul>
23   * 
24   * <pre class="highlight"><code class="java">
25   * {@literal @CsvBean(header=true, validateHeader=true)}
26   * {@literal @CsvPartial}(columnSize=5, headers={
27   *     {@literal @CsvPartial.Header(number=3, label="電話番号")},
28   *     {@literal @CsvPartial.Header(number=5, label="生年月日")})
29   * public class SampleCsv {
30   *     
31   *     {@literal @CsvColumn(number=1)}
32   *     private int id;
33   *
34   *     {@literal @CsvColumn(number=2, label="氏名")}
35   *     private String name;
36   *
37   *     // カラム番号3は読み込まない場合は、定義を行いません。
38   *
39   *     {@literal @CsvColumn(number=4, label="メールアドレス")}
40   *     private String email;
41   *
42   *     // カラム番号5はマッピングしない場合は、定義を行いません。
43   *     
44   *     // getter/setterは省略
45   * }
46   * </code></pre>
47   * 
48   * @since 2.0
49   * @author T.TSUCHIE
50   *
51   */
52  @Target({ElementType.TYPE})
53  @Retention(RetentionPolicy.RUNTIME)
54  @Documented
55  public @interface CsvPartial {
56      
57      /**
58       * 実際のCSVファイルのカラム数を指定します。
59       * <p>定義しているカラムの{@link CsvColumn#number()}より大きい値を指定します。</p>
60       * @return 1以上の値を指定します。
61       */
62      int columnSize();
63      
64      /**
65       * ヘッダー情報を定義するためのアノテーションです。
66       * @return Beanに定義されていないカラムのヘッダー情報を補足するために使用されます。
67       */
68      Header[] headers() default {};
69      
70      /**
71       * ヘッダー情報を表現します。
72       *
73       */
74      @interface Header {
75          
76          /**
77           * カラム番号を指定します。
78           * @return 1以上の値を指定します。
79           */
80          int number();
81          
82          /**
83           * ヘッダーの値を指定します。
84           * @return ヘッダーの値を指定します。
85           */
86          String label();
87          
88          /**
89           * 固定長を指定します。
90           * 複数指定した場合は先頭のアノテーションが採用されます。
91           * 
92           * @since 2.5
93           * @return 固定長カラムの設定。
94           */
95          CsvFixedSize[] fixedSize() default {};
96          
97      }
98  }
