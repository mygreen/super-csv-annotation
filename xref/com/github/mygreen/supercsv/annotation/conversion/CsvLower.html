17   * 
18   * <ul>

21   *   </li>
22   * </ul>
23   * 
24   * <pre class="highlight"><code class="java">
25   * {@literal @CsvBean}
26   * public class SampleCsv {
27   *     
28   *     {@literal @CsvColumn(number=1)}
29   *     {@literal @CsvLower}
30   *     private Integer name;
31   *     
32   *     // 書き込み時のみ適用する場合
33   *     {@literal @CsvColumn(number=2)}
34   *     {@literal @CsvLower(cases=BuildCase.Write)}
35   *     private String comment;
36   *     
37   *     // getter/setterは省略
38   * }
39   * </code></pre>
40   * 
41   * @since 2.0
42   * @author T.TSUCHIE
43   *
44   */
45  @Target({ElementType.FIELD, ElementType.ANNOTATION_TYPE})
46  @Retention(RetentionPolicy.RUNTIME)

50  public @interface CsvLower {
51      
52      /**
53       * 適用するケースを指定します。
54       * @return 何も指定しない場合は全てのケースに適用されます。

56      BuildCase[] cases() default {};
57      
58      /**
59       * グループのクラスを指定します。
60       * <p>処理ごとに適用するアノテーションを切り替えたい場合に指定します。
61       * @return 指定しない場合は、{@link DefaultGroup}が適用され全ての処理に適用されます。
62       */
63      Class<?>[] groups() default {};
64
