10  import java.util.regex.Pattern;
11  
12  import com.github.mygreen.supercsv.annotation.DefaultGroup;
13  import com.github.mygreen.supercsv.annotation.PatternFlag;
14  import com.github.mygreen.supercsv.builder.BuildCase;

16  /**
17   * 正規表現に一致した文字列を置換するためのアノテーションです。
18   * 
19   * <h3 class="description">基本的な使い方</h3>
20   * 
21   * <ul>
22   *   <li>属性 {@link #regex()}で、正規表現を指定します。
23   *     <br>{@link Pattern}で解釈可能な値を指定する必要があります。。
24   *   </li>
25   *   <li>属性{@link #replacement()}で置換後の値を指定します。</li>
26   *   <li>属性{@link #flags()}で、正規表現をコンパイルする際のフラグを列挙型{@link PatternFlag}で指定します。
27   *     <br>複数指定すると、ビットOR演算した結果と同じ意味になります。
28   *   </li>
29   * </ul>
30   * 
31   * <pre class="highlight"><code class="java">
32   * {@literal @CsvBean}
33   * public class SampleCsv {
34   *     
35   *     // 日時の「2016/12/17」の形式を「2016-12-17」の形式に変換します。
36   *     {@literal @CsvColumn(number=1)}
37   *     {@literal @CsvRegexReplace(regex="([0-9]{4})/([0-9]{1,2})/([0-9]{1,2})", replacement="$1-$2-$3")}
38   *     private LocalDate date;
39   *     
40   *     // コンパイル用のフラグを指定する場合
41   *     {@literal @CsvColumn(number=2)}
42   *     {@literal @CsvRegexReplace(regex="^comment(.+)", replacement='$1', flags={PatternFlag.CASE_INSENSITIVE})}
43   *     private String comment;
44   *     
45   *     
46   *     // getter/setterは省略
47   * }
48   * </code></pre>
49   * 
50   * @version 2.2
51   * @since 2.0
52   * @author T.TSUCHIE
53   *
54   */
55  @Target({ElementType.FIELD, ElementType.ANNOTATION_TYPE})
56  @Retention(RetentionPolicy.RUNTIME)
57  @Documented
58  @Repeatable(CsvRegexReplace.List.class)
59  @CsvConversion(value={})
60  public @interface CsvRegexReplace {
61      
62      /**
63       * 正規表現の値を指定します。
64       * <p>{@link Pattern}で解釈可能な値を指定してください。
65       * @return 正規表現の値。
66       */
67      String regex();
68      
69      /**
70       * 置換後の値を指定します。
71       * <p>一致した場合に入れ替える値となります。
72       * @return 置換後の値。
73       */
74      String replacement();
75      
76      /**
77       * 正規表現をコンパイルする際のフラグを指定します。
78       * <p>列挙型{@link PatternFlag}で指定します。</p>
79       * @return 正規表現のコンパイルする際のフラグ。
80       */
81      PatternFlag[] flags() default {};
82      
83      /**
84       * 正規表現で一致するものを検索するときに、部分一致で検索するかどうか。
85       * <ul>
86       *  <li>部分一致のとき、{@link Matcher#find()} を使用する。</li>
87       *  <li>全体一致のとき、{@link Matcher#matches()} を使用する。</li>
88       * </ul>
89       * 
90       * @since 2.2
91       * @return trueのとき部分一致。falseのとき全体一致。
92       */
93      boolean partialMatched() default false;
94      
95      /**
96       * 適用するケースを指定します。
97       * @return 何も指定しない場合は全てのケースに適用されます。
98       */
99      BuildCase[] cases() default {};
100     
101     /**
102      * グループのクラスを指定します。
103      * <p>処理ごとに適用するアノテーションを切り替えたい場合に指定します。
104      * @return 指定しない場合は、{@link DefaultGroup}が適用され全ての処理に適用されます。
105      */
106     Class<?>[] groups() default {};
107     
108     /**
109      * アノテーションの処理順序の定義。
110      * @return 値が大きいほど後に実行されます。
111      *         値が同じ場合は、アノテーションのクラス名の昇順になります。
112      */
113     int order() default 0;
114     
115     /**
116      * アノテーションを複数個指定する際の要素です。
117      */
118     @Target({ElementType.FIELD, ElementType.ANNOTATION_TYPE})
119     @Retention(RetentionPolicy.RUNTIME)
120     @Documented
121     @interface List {
122         
123         CsvRegexReplace[] value();
124     }
125     
126 }
