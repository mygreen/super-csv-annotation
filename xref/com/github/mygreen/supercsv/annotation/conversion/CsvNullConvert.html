8   import java.lang.annotation.Target;
9   
10  import com.github.mygreen.supercsv.annotation.DefaultGroup;
11  import com.github.mygreen.supercsv.builder.BuildCase;
12  
13  /**
14   * 指定した値と一致したときにnullに変換するためのアノテーションです。
15   * 
16   * <h3 class="description">基本的な使い方</h3>
17   * 
18   * <ul>
19   *   <li>属性{@link #value()}で、変換対象の値を指定します。
20   *   <br>複数指定可能です。
21   *   </li>
22   *   <li>比較する際に、大文字・小文字の区別を付けたくない場合は、属性{@link #ignoreCase()}の値を{@literal true}にします。</li>
23   *   <li>アノテーション{@link CsvDefaultValue}と併用する際には、処理結果が互いに変換対象となるため、
24   *      属性{@link #cases()}や{@link #groups()}で適用するケースを分けるようにしてください。
25   *   </li>
26   * </ul>
27   * 
28   * <pre class="highlight"><code class="java">
29   * {@literal @CsvBean}
30   * public class SampleCsv {
31   *     
32   *     // 複数指定可能です
33   *     {@literal @CsvColumn(number=1)}
34   *     {@literal @CsvNullConvert({"-", "N/A"})}
35   *     private Integer value;
36   *     
37   *     // 大文字・小文字の区別を行わない場合
38   *     {@literal @CsvColumn(number=2)}
39   *     {@literal @CsvNullConvert(value={"-", "N/A"}, ignoreCase=true)}
40   *     private Integer salary;
41   *     
42   *     // 読み込み時のみ適用する場合
43   *     {@literal @CsvColumn(number=3)}
44   *     {@literal @CsvNullConvert(value="-", cases=BuildCase.Read)}
45   *     private String comment;
46   *     
47   *     // getter/setterは省略
48   * }
49   * </code></pre>
50   * 
51   * @since 2.0
52   * @author T.TSUCHIE
53   *
54   */
55  @Target({ElementType.FIELD, ElementType.ANNOTATION_TYPE})
56  @Retention(RetentionPolicy.RUNTIME)
57  @Documented
58  @Repeatable(CsvNullConvert.List.class)
59  @CsvConversion(value={})
60  public @interface CsvNullConvert {
61      
62      /**
63       * 変換対象の値を指定します。
64       * @return 変換対象となる値を指定します。
65       * 
66       */
67      String[] value();
68      
69      /**
70       * 値を比較する際に大文字・小文字の区別を行わないかどうか指定します。
71       * @return trueの場合、大文字・小文字の区別を行いません。
72       */
73      boolean ignoreCase() default false;
74      
75      /**
76       * 適用するケースを指定します。
77       * @return 何も指定しない場合は全てのケースに適用されます。
78       */
79      BuildCase[] cases() default {};
80      
81      /**
82       * グループのクラスを指定します。
83       * <p>処理ごとに適用するアノテーションを切り替えたい場合に指定します。
84       * @return 指定しない場合は、{@link DefaultGroup}が適用され全ての処理に適用されます。
85       */
86      Class<?>[] groups() default {};
87      
88      @Target({ElementType.FIELD})
89      @Retention(RetentionPolicy.RUNTIME)
90      @Documented
91      @interface List {
92          
93          CsvNullConvert[] value();
94      }
95      
96  }
