12  /**
13   * 変換のアノテーションを表現するためのメタアノテーションです。
14   * <p>変換のアノテーションには必ず付与します。</p>
15   * 
16   * <h3 class="description">基本的な使い方</h3>
17   * <ul>
18   *  <li>{@literal @Target}として、{@link ElementType#FIELD}と{@link ElementType#ANNOTATION_TYPE}の2つを指定します。
19   *     <br>アノテーションを合成する際にはアノテーションにも付与するため、{@link ElementType#ANNOTATION_TYPE}を追加しておきます。
20   *  </li>
21   *  <li>繰り返しのアノテーションとして利用できるよう {@literal @Repeatable}を付与します。
22   *     <br>繰り返しのアノテーションを格納する内部アノテーションとして、{@literal List}を定義します。
23   *  </li>
24   *  <li>検証用のアノテーションであることを示すためのメタアノテーション {@literal @CsvContraint}を指定します。
25   *    <br>属性{@link #value()}で、{@link ConversionProcessorFactory}の実装クラスを指定します。
26   *  </li>
27   *  <li>共通の属性として、{@literal cases, groups, order}を定義します。
28   *    <br>省略した場合は、それぞれのデフォルト値が適用されます。
29   *  </li>
30   *  <li>必要であれば、固有の属性を定義します。</li>
31   * </ul>
32   * 
33   * <pre class="highlight"><code class="java">
34   * // 独自の値の検証用のアノテーション
35   * {@literal @Target({ElementType.FIELD, ElementType.ANNOTATION_TYPE})}
36   * {@literal @Retention(RetentionPolicy.RUNTIME)}
37   * {@literal @Documented}
38   * {@literal @Repeatable(CsvCustomConversion.List.class)}
39   * {@literal @CsvConstraint(CustomConversionFactory.class)}  // ファクトリクラスを指定
40   * public {@literal @interface} CsvCustomConversion {
41   *     
42   *     // 固有の属性 - 必要であれば定義します。
43   *     String text();
44   *     
45   *     // 共通の属性 - ケース
46   *     BuildCase[] cases() default {};
47   *    
48   *     // 共通の属性 - グループ
49   *     {@literal Class<?>[]} groups() default {};
50   *     
51   *     // 共通の属性 - 並び順
52   *     int order() default 0;
53   *     
54   *     // 繰り返しのアノテーションの格納用アノテーションの定義
55   *     {@literal @Target({ElementType.FIELD, ElementType.ANNOTATION_TYPE})}
56   *     {@literal @Retention(RetentionPolicy.RUNTIME)}
57   *     {@literal @Documented}
58   *     {@literal @interface} List {
59   *     
60   *         CsvCustomConversion[] value();
61   *     }
62   * }
63   * </code></pre>
64   * 
65   * 
66   * @since 2.0
67   * @author T.TSUCHIE
68   *
69   */
70  @Target({ElementType.ANNOTATION_TYPE})
71  @Retention(RetentionPolicy.RUNTIME)
72  @Documented
73  public @interface CsvConversion {
74      
75      /**
76       * アノテーションに対応したCellProcessorを作成するファクトリクラスを指定します。
77       * <p>省略した場合、{@link AbstractProcessorBuilder#registerForConversion(Class, ConversionProcessorFactory)}で手動で登録する必要があります。</p>
78       * @return {@link ConversionProcessorFactory}を実装したクラスを指定します。
79       */
80      Class<? extends ConversionProcessorFactory<?>>[] value();
81      
82  }
