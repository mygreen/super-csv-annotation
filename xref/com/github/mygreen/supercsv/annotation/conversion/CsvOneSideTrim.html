4   import java.lang.annotation.ElementType;
5   import java.lang.annotation.Repeatable;
6   import java.lang.annotation.Retention;
7   import java.lang.annotation.RetentionPolicy;
8   import java.lang.annotation.Target;
9   
10  import com.github.mygreen.supercsv.annotation.DefaultGroup;
11  import com.github.mygreen.supercsv.builder.BuildCase;
12  
13  /**
14   * 片方だけトリムするためのアノテーションです。
15   * 
16   * <h3 class="description">基本的な使い方</h3>
17   * 
18   * <ul>
19   *   <li>処理対象の値がnullの場合は、変換は行いません。</li>
20   *   <li>文字列以外の数値型に対して付与した場合、読み込み時にトリミングの結果として空文字となった場合は、
21   *       値が空として処理されます。
22   *       <br>例えば、Integer型にマッピングする際にはnullとなります。</li>
23   * </ul>
24   * 
25   * <pre class="highlight"><code class="java">
26   * {@literal @CsvBean}
27   * public class SampleCsv {
28   *     
29   *     // デフォルトは、右側の半角文字をトリムします。
30   *     {@literal @CsvColumn(number=1)}
31   *     {@literal @CsvOneSideTrim}
32   *     private Integer value;
33   *     
34   *     // 左側の全角空白をトリムする場合。
35   *     {@literal @CsvColumn(number=2)}
36   *     {@literal @CsvOneSideTrim(trimChar='　', leftAlign=true)}
37   *     private String comment;
38   *     
39   *     // getter/setterは省略
40   * }
41   * </code></pre>
42   * 
43   * @since 2.1
44   * @author T.TSUCHIE
45   *
46   */
47  @Target({ElementType.FIELD, ElementType.ANNOTATION_TYPE})
48  @Retention(RetentionPolicy.RUNTIME)
49  @Documented
50  @Repeatable(CsvOneSideTrim.List.class)
51  @CsvConversion(value={})
52  public @interface CsvOneSideTrim {
53      
54      /**
55       * トリミング対象の文字を指定します。
56       *
57       * @return トリミング対象の文字。
58       */
59      char trimChar() default ' ';
60      
61      /**
62       * 左側をトリムするかどうか指定します。
63       *
64       * @return falseの場合は右側をトリムします。
65       */
66      boolean leftAlign() default false;
67      
68      /**
69       * 適用するケースを指定します。
70       * @return 何も指定しない場合は全てのケースに適用されます。
71       */
72      BuildCase[] cases() default {};
73      
74      /**
75       * グループのクラスを指定します。
76       * <p>処理ごとに適用するアノテーションを切り替えたい場合に指定します。
77       * @return 指定しない場合は、{@link DefaultGroup}が適用され全ての処理に適用されます。
78       */
79      Class<?>[] groups() default {};
80      
81      /**
82       * アノテーションの処理順序の定義。
83       * @return 値が大きいほど後に実行されます。
84       *         値が同じ場合は、アノテーションのクラス名の昇順になります。
85       */
86      int order() default 0;
87      
88      @Target({ElementType.FIELD, ElementType.ANNOTATION_TYPE})
89      @Retention(RetentionPolicy.RUNTIME)
90      @Documented
91      @interface List {
92          
93          CsvOneSideTrim[] value();
94      }
95  }
