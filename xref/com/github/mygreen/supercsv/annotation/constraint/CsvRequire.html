15  /**
16   * 値が必須かどうか検証するためのアノテーションです。
17   * <p>全ての型に指定可能です。</p>
18   * 
19   * <h3 class="description">基本的な使い方</h3>
20   * <ul>
21   *   <li>オブジェクト型の場合、nullでないかどうか検証します。
22   *     <br>プリミティブ型のように必ず初期値が存在する場合は、エラーとなりません。
23   *   </li>
24   *   <li>文字列型の場合、属性{@link #considerEmpty()}、{@link #considerBlank()}を指定することで、
25   *     空文字、空白文字かどうか判定することができます。
26   *     <ul>
27   *       <li>属性{@link #considerEmpty()}で、長さが0の空文字も考慮するか指定します。
28   *         <br>trueの場合、空文字でもエラーとなります。</li>
29   *       <li>属性{@link #considerBlank()}で、半角スペースのみの空白文字も考慮するか指定します。
30   *         <br>trueの場合、空白文字でもエラーとなります。
31   *       </li>
32   *     </ul>
33   *   </li>
34   * </ul>
35   * 
36   * 
37   * <pre class="highlight"><code class="java">
38   * {@literal @CsvBean}
39   * public class SampleCsv {
40   *     
41   *     {@literal @CsvColumn(number=1)}
42   *     {@literal @CsvRequire}
43   *     private Integer id;
44   *     
45   *     {@literal @CsvColumn(number=2)}
46   *     {@literal @CsvRequire(considerBlank=true)}
47   *     private String name;
48   *     
49   *     // getter/setterは省略
50   * }
51   * </code></pre>
52   * 
53   * @since 2.0
54   * @author T.TSUCHIE
55   *
56   */
57  @Target({ElementType.FIELD, ElementType.ANNOTATION_TYPE})
58  @Retention(RetentionPolicy.RUNTIME)
59  @Documented
60  @Repeatable(CsvRequire.List.class)
61  @CsvConstraint(value={})
62  public @interface CsvRequire {
63      
64      /**
65       * 空文字（長さが0）の時も考慮するかどうか指定します。
66       * @return trueに設定すると、値がnullまたは空文字のときもエラーとなります。
67       */
68      boolean considerEmpty() default false;
69      
70      /**
71       * 空白文字の時も考慮するかどうかを指定します。
72       * @return trueに設定すると、値がnullまたは空文字、空白文字のときもエラーとなります。
73       */
74      boolean considerBlank() default false;
75      
76      /**
77       * エラー時のメッセージを指定します。
78       * <p>{@literal {key}}の書式の場合、プロパティファイルから取得した値を指定できます。</p>
79       * 
80       * <p>使用可能なメッセージ中の変数は下記の通りです。</p>
81       * <ul>
82       *   <li>lineNumber : カラムの値に改行が含まれている場合を考慮した実際の行番号です。1から始まります。</li>
83       *   <li>rowNumber : CSVの行番号です。1から始まります。</li>
84       *   <li>columnNumber : CSVの列番号です。1から始まります。</li>
85       *   <li>label : カラムの見出し名です。</li>
86       *   <li>validatedValue : 実際のカラムの値です。</li>
87       *   <li>considerEmpty : アノテーションの属性{@link #considerEmpty()}の値</li>
88       *   <li>considerBlank : アノテーションの属性{@link #considerBlank()}の値</li>
89       * </ul>
90       * 
91       * @return 省略した場合は、適用された{@link CellProcessor}に基づいたメッセージが出力されます。
92       */
93      String message() default "{com.github.mygreen.supercsv.annotation.constraint.CsvRequire.message}";
94      
95      /**
96       * 適用するケースを指定します。
97       * @return 何も指定しない場合は全てのケースに適用されます。
98       */
99      BuildCase[] cases() default {};
100     
101     /**
102      * グループのクラスを指定します。
103      * <p>処理ごとに適用するアノテーションを切り替えたい場合に指定します。
104      * @return 指定しない場合は、{@link DefaultGroup}が適用され全ての処理に適用されます。
105      */
106     Class<?>[] groups() default {};
107     
108     /**
109      * アノテーションの処理順序の定義。
110      * @return 値が大きいほど後に実行されます。
111      *         値が同じ場合は、アノテーションのクラス名の昇順になります。
112      */
113     int order() default Integer.MIN_VALUE;

116      * アノテーションを複数個指定する際の要素です。
117      */
118     @Target({ElementType.FIELD, ElementType.ANNOTATION_TYPE})
