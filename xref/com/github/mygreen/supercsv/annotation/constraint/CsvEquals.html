4   import java.lang.annotation.ElementType;
5   import java.lang.annotation.Repeatable;
6   import java.lang.annotation.Retention;
7   import java.lang.annotation.RetentionPolicy;
8   import java.lang.annotation.Target;
9   import java.util.Collection;
10  
11  import org.supercsv.cellprocessor.ift.CellProcessor;
12  
13  import com.github.mygreen.supercsv.annotation.DefaultGroup;
14  import com.github.mygreen.supercsv.builder.BuildCase;
15  import com.github.mygreen.supercsv.cellprocessor.constraint.EqualedValueProvider;
16  import com.github.mygreen.supercsv.cellprocessor.format.TextPrinter;
17  
18  /**
19   * セルの値が指定した値と等しいかどうか検証するためのアノテーションです。
20   * <p>全てのクラスタイプに指定可能です。</p>
21   * 
22   * <h3 class="description">基本的な使い方</h3>
23   * <p>属性{@link #value()}で比較対象の値を指定します。</p>
24   * <ul>
25   *   <li>値は複数指定可能で、その場合、何れかの値に一致すれば問題ありません。</li>
26   *   <li>数値や日時など書式を持つ場合は、その書式に沿った値を指定する必要があります。</li>
27   * </ul>
28   * 
29   * <pre class="highlight"><code class="java">
30   * {@literal @CsvBean}
31   * public class SampleCsv {
32   *     
33   *     {@literal @CsvColumn(number=1)}
34   *     {@literal @CsvEquals({"admin", "normal"})}
35   *     private String type;
36   *     
37   *     // 書式を指定している場合
38   *     {@literal @CsvColumn(number=2)}
39   *     {@literal @CsvNumberForamt(pattern="#,##0")}
40   *     {@literal @CsvEquals({"-1,000", "1,000"})}
41   *     private Integer number;
42   *     
43   *     // getter/setterは省略
44   * }
45   * </code></pre>
46   * 
47   * <h3 class="description">DBやファイルなどのリソースから取得する場合</h3>
48   * <p>値をDBやファイルなどの別リソースから取得する場合は、属性{@link #provider()}にて、
49   *    プロバイダ{@link EqualedValueProvider}の実装クラスを指定します。
50   * </p>
51   * <p>Spring Frameworkと連携している場合は、プロバイダクラスをSpringBeanとして登録しておくことでインジェクションできます。</p>
52   * 
53   * <pre class="highlight"><code class="java">
54   * {@literal @CsvBean}
55   * public class SampleCsv {
56   *     
57   *     {@literal @CsvColumn(number=1)}
58   *     {@literal @CsvEquals(provider=FileEqualedValueProvider.class)}
59   *     private Integer number;
60   *     
61   *     // setter/getterは省略
62   * }
63   * 
64   * // プロバイダクラスの実装（ファイルから値を取得する）
65   * public class FileEqualedValueProvider implements {@literal EqualedValueProvider<Integer>} {
66   *     
67   *     {@literal @Override}
68   *     public {@literal Collection<Integer>} getEqualedValues(final FieldAccessor field) {
69   *         
70   *         String lines;
71   *         try {
72   *              lines = Files.readAllLines(
73   *                      new File("equaled_value.txt").toPath(), Charset.forName("UTF-8"));
74   *              
75   *         } catch (IOException e) {
76   *             throw new RuntimeException("fail reading the equaled value file.", e);
77   *         }
78   *         
79   *         // 読み取った値をInteger型に変換します。
80   *         return lines.stream()
81   *             .map(l {@literal ->} Integer.valueOf(l))
82   *             .collect(Collectors.toList());
83   *         
84   *     }
85   * }
86   * </code></pre>
87   * 
88   * 
89   * @since 2.0
90   * @author T.TSUCHIE
91   *
92   */
93  @Target({ElementType.FIELD, ElementType.ANNOTATION_TYPE})
94  @Retention(RetentionPolicy.RUNTIME)
95  @Documented
96  @Repeatable(CsvEquals.List.class)
97  @CsvConstraint(value={})
98  public @interface CsvEquals {
99      
100     /**
101      * 比較する値を指定します。
102      * <p>複数指定した場合は、何れかに一致すれば問題ありません。</p>
103      * <p>ブール型、数値や日時型の場合は、アノテーションで指定した書式に沿った値を指定する必要があります。</p>
104      * @return 比較対象の値。
105      */
106     String[] value() default {};
107     
108     /**
109      * 値を取得するプロバイダクラスを指定します。。
110      * @return {@link EqualedValueProvider}の実装クラス。
111      */
112     @SuppressWarnings("rawtypes")
113     Class<? extends EqualedValueProvider> [] provider() default {};
114     
115     
116     /**
117      * エラー時のメッセージを指定します。
118      * <p>{@literal {key}}の書式の場合、プロパティファイルから取得した値を指定できます。</p>
119      * 
120      * <p>使用可能なメッセージ中の変数は下記の通りです。</p>
121      * <ul>
122      *   <li>lineNumber : カラムの値に改行が含まれている場合を考慮した実際の行番号です。1から始まります。</li>
123      *   <li>rowNumber : CSVの行番号です。1から始まります。</li>
124      *   <li>columnNumber : CSVの列番号です。1から始まります。</li>
125      *   <li>label : カラムの見出し名です。</li>
126      *   <li>validatedValue : 実際のカラムの値です。</li>
127      *   <li>equalsValues : アノテーションの属性{@link #value()}をフィールドの型に変換した値です。{@link Collection}の形式です。</li>
128      *   <li>formatter : カラムの値に対数するフォーマッタです。{@link TextPrinter#print(Object)}でvalidatedValue, equalsValuesの値を文字列に変換します。</li>
129      * </ul>
130      * 
131      * @return 省略した場合は、適用された{@link CellProcessor}に基づいたメッセージが出力されます。
132      */
133     String message() default "{com.github.mygreen.supercsv.annotation.constraint.CsvEquals.message}";
134     
135     /**
136      * 適用するケースを指定します。
137      * @return 何も指定しない場合は全てのケースに適用されます。
138      */
139     BuildCase[] cases() default {};
140     
141     /**
142      * グループのクラスを指定します。
143      * <p>処理ごとに適用するアノテーションを切り替えたい場合に指定します。
144      * @return 指定しない場合は、{@link DefaultGroup}が適用され全ての処理に適用されます。
145      */
146     Class<?>[] groups() default {};
147     
148     /**
149      * アノテーションの処理順序の定義。
150      * @return 値が大きいほど後に実行されます。
151      *         値が同じ場合は、アノテーションのクラス名の昇順になります。
152      */
153     int order() default 0;
154     
155     /**
156      * アノテーションを複数個指定する際の要素です。
157      */
158     @Target({ElementType.FIELD, ElementType.ANNOTATION_TYPE})
159     @Retention(RetentionPolicy.RUNTIME)
160     @Documented
161     @interface List {
162         
163         CsvEquals[] value();
164     }
165 }
