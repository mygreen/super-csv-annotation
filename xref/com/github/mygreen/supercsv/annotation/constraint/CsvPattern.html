3   import java.lang.annotation.Documented;
4   import java.lang.annotation.ElementType;
5   import java.lang.annotation.Repeatable;
6   import java.lang.annotation.Retention;
7   import java.lang.annotation.RetentionPolicy;
8   import java.lang.annotation.Target;
9   import java.util.regex.Pattern;
10  
11  import org.supercsv.cellprocessor.ift.CellProcessor;
12  
13  import com.github.mygreen.supercsv.annotation.DefaultGroup;
14  import com.github.mygreen.supercsv.annotation.PatternFlag;
15  import com.github.mygreen.supercsv.builder.BuildCase;
16  
17  /**
18   * 値が指定した正規表現に一致するかどうか検証するためのアノテーションです。
19   * <p>文字列型に指定可能です。</p>
20   * 
21   * <h3 class="description">基本的な使い方</h3>
22   * 
23   * <ul>
24   *   <li>属性 {@link #regex()}で、正規表現を指定します。
25   *     <br>{@link Pattern}で解釈可能な値を指定する必要があります。。
26   *   </li>
27   *   <li>属性{@link #flags()}で、正規表現をコンパイルする際のフラグを列挙型{@link PatternFlag}で指定します。
28   *     <br>複数指定すると、ビットOR演算した結果と同じ意味になります。
29   *   </li>
30   * </ul>
31   * 
32   * <pre class="highlight"><code class="java">
33   * {@literal @CsvBean}
34   * public class SampleCsv {
35   *     
36   *     {@literal @CsvColumn(number=1)}
37   *     {@literal @CsvPattern(regex="([0-9]{4})/([0-9]{1,2})/([0-9]{1,2})")}
38   *     private LocalDate date;
39   *     
40   *     // コンパイル用のフラグを指定する場合
41   *     {@literal @CsvColumn(number=2)}
42   *     {@literal @CsvPattern(regex="^comment(.+)", flags={PatternFlag.CASE_INSENSITIVE})}
43   *     private String comment;
44   *     
45   *     
46   *     // getter/setterは省略
47   * }
48   * </code></pre>
49   *
50   * @since 2.0
51   * @author T.TSUCHIE
52   *
53   */
54  @Target({ElementType.FIELD, ElementType.ANNOTATION_TYPE})
55  @Retention(RetentionPolicy.RUNTIME)
56  @Documented
57  @Repeatable(CsvPattern.List.class)
58  @CsvConstraint(value={})
59  public @interface CsvPattern {
60      
61      /**
62       * 正規表現の値。
63       * @return {@link Pattern}で解釈可能な値。
64       */
65      String regex();
66      
67      /**
68       * 正規表現をコンパイルする際のフラグを指定します。
69       * @return 列挙型{@link PatternFlag}で指定します。
70       */
71      PatternFlag[] flags() default {};
72      
73      /**
74       * 正規表現の説明。
75       * @return エラー時のメッセージ中の変数として使用可能。
76       */
77      String description() default "";
78      
79      /**
80       * エラー時のメッセージを指定します。
81       * <p>{@literal {key}}の書式の場合、プロパティファイルから取得した値を指定できます。</p>
82       * 
83       * <p>使用可能なメッセージ中の変数は下記の通りです。</p>
84       * <ul>
85       *   <li>lineNumber : カラムの値に改行が含まれている場合を考慮した実際の行番号です。1から始まります。</li>
86       *   <li>rowNumber : CSVの行番号です。1から始まります。</li>
87       *   <li>columnNumber : CSVの列番号です。1から始まります。</li>
88       *   <li>label : カラムの見出し名です。</li>
89       *   <li>validatedValue : 実際のカラムの値です。</li>
90       *   <li>regex : アノテーションの属性{@link #regex()}の値です。</li>
91       *   <li>flags : アノテーションの属性{@link #flags()}の値をint型に変換した値です。</li>
92       *   <li>description : アノテーションの属性{@link #description()}の値です。</li>
93       * </ul>
94       * 
95       * 
96       * @return 省略した場合は、適用された{@link CellProcessor}に基づいたメッセージが出力されます。
97       */
98      String message() default "{com.github.mygreen.supercsv.annotation.constraint.CsvPattern.message}";
99      
100     /**
101      * 適用するケースを指定します。
102      * @return 何も指定しない場合は全てのケースに適用されます。
103      */
104     BuildCase[] cases() default {};
105     
106     /**
107      * グループのクラスを指定します。
108      * <p>処理ごとに適用するアノテーションを切り替えたい場合に指定します。
109      * @return 指定しない場合は、{@link DefaultGroup}が適用され全ての処理に適用されます。
110      */
111     Class<?>[] groups() default {};
112     
113     /**
114      * アノテーションの処理順序の定義。
115      * @return 値が大きいほど後に実行されます。
116      *         値が同じ場合は、アノテーションのクラス名の昇順になります。
117      */
118     int order() default 0;
119     
120     /**
121      * アノテーションを複数個指定する際の要素です。
122      */
123     @Target({ElementType.FIELD, ElementType.ANNOTATION_TYPE})
124     @Retention(RetentionPolicy.RUNTIME)
125     @Documented
126     @interface List {
127         
128         CsvPattern[] value();
129     }
130 }
