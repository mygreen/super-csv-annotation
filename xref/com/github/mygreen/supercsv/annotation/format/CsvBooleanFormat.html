3   import java.lang.annotation.Documented;
4   import java.lang.annotation.ElementType;
5   import java.lang.annotation.Retention;
6   import java.lang.annotation.RetentionPolicy;
7   import java.lang.annotation.Target;
8   import java.util.LinkedHashSet;
9   
10  import org.supercsv.cellprocessor.ift.CellProcessor;
11  
12  
13  /**
14   * boolean/Boolean型の書式を定義するためのアノテーションです。
15   * 
16   * <h3 class="description">基本的な使い方</h3>
17   * 
18   * <ul>
19   *   <li>属性{@link #readForTrue()}、{@link #readForFalse()}で、読み込み時のtrueまたはfalseと判定する候補の値を指定します。</li>
20   *   <li>属性{@link #writeAsTrue()}、{@link #writeAsFalse()}で、書き込み時のtrueまたはfalse値に該当する文字を指定します。</li>
21   *   <li>属性{@link #ignoreCase()}で、読み込み時に大文字・小文字の区別なく候補の値と比較するか指定します。</li>
22   * </ul>
23   * 
24   * <pre class="highlight"><code class="java">
25   * {@literal @CsvBean}
26   * public class SampleCsv {
27   *     
28   *     // boolean型の読み込み時のtrueとfalseの値の変換規則を指定します。
29   *     {@literal @CsvColumn(number=1)}
30   *     {@literal @CsvBooleanFormat(}
31   *           readForTrue={"○", "有効", "レ"}, readForFalse={"×", "無効", "-", ""},
32   *           writeAsTrue="○", writeAsFalse="×")
33   *     private boolean availaled;
34   *     
35   *     // 読み込み時の大文字・小文字の区別を行わない
36   *     {@literal @CsvColumn(number=2, label="チェック")}
37   *     {@literal @CsvBooleanFormat(readForTrue={"OK"}, readForFalse={"NO"}, ignoreCase=true)}
38   *     private Boolean checked;
39   *     
40   *     // getter/setterは省略
41   * }
42   * </code></pre>
43   * 
44   * @version 2.0
45   * @author T.TSUCHIE
46   *
47   */
48  @Target({ElementType.FIELD, ElementType.ANNOTATION_TYPE})
49  @Retention(RetentionPolicy.RUNTIME)
50  @Documented
51  public @interface CsvBooleanFormat {
52      
53      /**
54       * 読み込み時に{@literal true} とする候補を指定します。
55       * <p>{@link #readForFalse()} と重複する値がある場合は、{@link #readForTrue()} の値が優先されます。</p>
56       * 
57       * @return trueとして読み込む候補。
58       */
59      String[] readForTrue() default {"true", "1", "yes", "on", "y", "t"};
60      
61      /**
62       * 読み込み時に{@literal false} とする候補を指定します。
63       * <p>{@link #readForTrue()} と重複する値がある場合は、{@link #readForTrue()} の値が優先されます。</p>
64       * 
65       * @return falseとして読み込む候補。
66       */
67      String[] readForFalse() default {"false", "0", "no", "off", "f", "n"};
68      
69      /**
70       * 書き込み時に{@literal true}の値を表現する値を指定します。
71       * <p>`true`以外の`○`など他の値として出力したい場合に指定します。</p>
72       * 
73       * <p>使用可能なメッセージ中の変数は下記の通りです。</p>
74       * <ul>
75       *   <li>lineNumber : カラムの値に改行が含まれている場合を考慮した実際の行番号です。1から始まります。</li>
76       *   <li>rowNumber : CSVの行番号です。1から始まります。</li>
77       *   <li>columnNumber : CSVの列番号です。1から始まります。</li>
78       *   <li>label : カラムの見出し名です。</li>
79       *   <li>validatedValue : 実際のカラムの値です。</li>
80       *   <li>ignoreCase : アノテーションの属性{@link #ignoreCase()}の値です。</li>
81       *   <li>failToFalse : アノテーションの属性{@link #failToFalse()}の値です。</li>
82       *   <li>trueValues : アノテーションの属性{@link #readForTrue()}を{@link LinkedHashSet}型にした値です。</li>
83       *   <li>falseValues : アノテーションの属性{@link #readForFalse()}を{@link LinkedHashSet}型にした値です。</li>
84       * </ul>
85       * 
86       * @return trueの値の代替として出力される値。
87       */
88      String writeAsTrue() default "true";
89      
90      /**
91       * 書き込み時に{@literal false}の値を表現する値を指定します。
92       * <p>`false`以外の`×`など他の値として出力したい場合に指定します。</p>
93       * 
94       * @return falseの値の代替として出力される値。
95       */
96      String writeAsFalse() default "false";
97      
98      /**
99       * 読み込み時に、大文字・小文字を区別なく候補の値と比較して判定するか指定します。
100      * @return trueの場合、大文字・小文字の区別は行いません。
101      */
102     boolean ignoreCase() default false;
103     
104     /**
105      * 読み込み時に {@link #readForTrue()}、{@link #readForFalse()} で指定した候補の値と一致しない場合、値をfalseとして読み込み込むか指定します。
106      * @return trueの場合、読み込み用の候補の値と一致しない場合、falseとして読み込みます。
107      */
108     boolean failToFalse() default false;
109     
110     /**
111      * エラー時のメッセージを指定します。
112      * <p>{@literal {key}}の書式の場合、プロパティファイルから取得した値を指定できます。</p>
113      * <p>このメッセージは、文字列をブール型にパースする際に失敗したときに適用されるメッセージに使用します。</p>
114      * @return 省略した場合は、適用された{@link CellProcessor}に基づいたメッセージが出力されます。
115      */
116     String message() default "{com.github.mygreen.supercsv.annotation.format.CsvBooleanFormat.message}";
117     
118 }
