121     /**
122      * 指定したヘッダーを元に、マッピング情報を補完する。
123      * <p>カラムの位置である番号を確定する。</p>
124      * <p>存在しないカラムがある場合は、部分的な読み込みとして、ダミーのカラム情報を作成する。</p>
125      * @param headers
126      * @param SuperCsvInvalidAnnotationException
127      */
128     private void setupMappingColumns(final String[] headers) {
129         
130         final List<ColumnMapping> columnMappingList = beanMapping.getColumns();
131         
132         // 一致するラベルがあれば、カラムの番号を補完する
133         final int headerSize = headers.length;
134         for(int i=0; i < headerSize;i ++) {
135             
136             final String header = headers[i];
137             
138             /*
139              * 番号が決まっておらず、ラベルが一致するカラム情報を抽出する。
140              * ※既に番号が決まっているが、ラベルが一致しないのものは、後からチェックする。
141              */
142             List<ColumnMapping> undeterminedColumnList = columnMappingList.stream()
143                 .filter(col -> !col.isDeterminedNumber())
144                 .filter(col -> col.getLabel().equals(header))
145                 .collect(Collectors.toList());
146             
147             final int columnNumber = i+1;
148             undeterminedColumnList.forEach(col -> col.setNumber(columnNumber));
149             
150         }
151         
152         // カラムの番号順に並び変える
153         columnMappingList.sort(null);
154         
155         /* 
156          * ヘッダーでは指定されていない番号が未決定のカラムについて、空いている番号に振る
157          * ・@CsvPartialで指定した場合とかぶらないようにすべき？
158          */
159         final Optional<CsvPartial> partialAnno = Optional.ofNullable(beanMapping.getType().getAnnotation(CsvPartial.class));
160         final List<ColumnMapping> undeterminedColumnList = columnMappingList.stream()
161                 .filter(col -> !col.isDeterminedNumber())
162                 .collect(Collectors.toList());
163         
164         if(!undeterminedColumnList.isEmpty()) {
165             final Set<Integer> determinedNumbers = columnMappingList.stream()
166                     .filter(col -> col.isDeterminedNumber())
167                     .map(col -> col.getNumber())
168                     .collect(Collectors.toSet());
169             
170             // @CsvParitalで定義されているカラムは、決定されている番号として処理する。
171             partialAnno.ifPresent(anno -> Arrays.stream(anno.headers()).forEach(header -> determinedNumbers.add(header.number())));
172             
173             int counter = 1;
174             for(ColumnMapping col : undeterminedColumnList) {
175                 while(true) {
176                     if(!determinedNumbers.contains(counter)) {
177                         break;
178                     }
179                     counter++;
180                 }
181                 
182                 col.setNumber(counter);
183                 determinedNumbers.add(counter);
184             }
185             
186             // 第度カラムの番号順に並び変える
187             columnMappingList.sort(null);
188         
189         }
190         
191         // 重複しているカラム番号のチェック
192         BeanMappingFactoryHelper.validateDuplicatedColumnNumber(beanMapping.getType(), columnMappingList);
193         
194         // 不足しているカラム番号の補完
195         BeanMappingFactoryHelper.supplyLackedNumberMappingColumn(beanMapping.getType(), columnMappingList, partialAnno, headers);
196         
197         beanMapping.setColumns(columnMappingList);
198         
199     }

231     /**
232      * レコードのデータを全て書き込みます。
233      * <p>ヘッダー行も自動的に処理されます。2回目以降に呼び出した場合、ヘッダー情報は書き込まれません。</p>
234      * 
235      * @param sources 書き込むレコードのデータ。
236      * @param continueOnError continueOnError レコードの処理中に、
237      *        例外{@link SuperCsvBindingException}が発生しても、続行するかどうか指定します。
238      *        trueの場合、例外が発生しても、次の処理を行います。
239      * @throws NullPointerException sources is null.
240      * @throws IOException レコードの出力に失敗した場合。
241      * @throws SuperCsvBindingException セルの値に問題がある場合
242      * @throws SuperCsvException 設定など、その他に問題がある場合
243      * 
244      */
245     public void writeAll(final Collection<T> sources, final boolean continueOnError) throws IOException {
246         
247         Objects.requireNonNull(sources, "sources should not be null.");
248         
249         if(!initialized) {
250             init();
251         }
252         
253         if(beanMappingCache.getOriginal().isHeader() && getLineNumber() == 0) {
254             writeHeader();
255         }
256         
257         for(T record : sources) {
258             try {
259                 write(record);
260             } catch(SuperCsvBindingException e) {
261                 if(!continueOnError) {
262                     throw e;
263                 }
264             }
265         }
266         
267         super.flush();
268         
269     }
