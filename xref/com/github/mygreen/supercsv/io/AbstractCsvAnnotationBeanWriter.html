75      /**
76       * レコードを書き込みます。
77       * 
78       * @param source 書き込むレコード。
79       * @throws NullPointerException source is null.
80       * @throws IOException レコードの出力に失敗した場合。
81       * @throws SuperCsvException レコードの値に問題がある場合
82       * 
83       */
84      public void write(final T source) throws IOException {
85          
86          Objects.requireNonNull(source, "the bean to write should not be null.");
87          
88          // update the current row/line numbers
89          super.incrementRowAndLineNo();
90          
91          final CsvContext context = new CsvContext(getLineNumber(), getRowNumber(), 1);
92          context.setRowSource(Collections.emptyList());  // 空の値を入れる
93          
94          final CsvBindingErrors bindingErrors = new CsvBindingErrors(beanMappingCache.getOriginal().getType());
95          
96          // コールバックメソッドの実行（書き込み前）
97          for(CallbackMethod callback : beanMappingCache.getOriginal().getPreWriteMethods()) {
98              callback.invoke(source, context, bindingErrors, beanMappingCache.getOriginal());
99          }
100         
101         // extract the bean values
102         extractBeanValues(source, beanMappingCache.getNameMapping());
103         context.setRowSource(new ArrayList<Object>(beanValues));
104         
105         Optional<SuperCsvRowException> rowException = Optional.empty();
106         try {
107             executeCellProcessors(processedColumns, beanValues, beanMappingCache.getCellProcessorsForWriting(), context);
108             
109         } catch(SuperCsvRowException e) {
110             /*
111              * カラムごとのCellProcessorのエラーの場合、別なValidatorで値を検証するために、
112              * 後から判定を行うようにする。
113              */
114             rowException = Optional.of(e);
115             
116             final List<CsvError> errors = exceptionConverter.convert(e, beanMappingCache.getOriginal());
117             bindingErrors.addAllErrors(errors);
118             
119         } catch(SuperCsvException e) {
120             // convert exception and format to message.
121             errorMessages.addAll(exceptionConverter.convertAndFormat(e, beanMappingCache.getOriginal()));
122             throw e;
123         }
124         
125         // レコード、Beanの入力値検証
126         if(!beanMappingCache.getOriginal().isSkipValidationOnWrite()) {
127             for(CsvValidator<T> validator : validators) {
128                 validator.validate(source, bindingErrors, new ValidationContext<>(context, beanMappingCache.getOriginal()));
129             }
130         }
131         
132         // エラーメッセージの変換
133         processErrors(bindingErrors, context, rowException);
134         
135         // write the list
136         super.writeRow(processedColumns);
137         
138         // コールバックメソッドの実行（書き込み後）
139         for(CallbackMethod callback : beanMappingCache.getOriginal().getPostWriteMethods()) {
140             callback.invoke(source, context, bindingErrors, beanMappingCache.getOriginal());
141         }
142         
143         // エラーメッセージの変換
144         processErrors(bindingErrors, context, rowException);
145         
146     }

148     /**
149      * 行の例外情報をメッセージに変換したりします。
150      * @param bindingErrors
151      * @param context
152      * @param rowException
153      */
154     protected void processErrors(final CsvBindingErrors bindingErrors, final CsvContext context,
155             final Optional<SuperCsvRowException> rowException) {
156         
157         if(bindingErrors.hasErrors()) {
158             final List<String> message = bindingErrors.getAllErrors().stream()
159                     .map(error -> error.format(exceptionConverter.getMessageResolver(), exceptionConverter.getMessageInterpolator()))
160                     .collect(Collectors.toList());
161             errorMessages.addAll(message);
162             
163             final SuperCsvBindingException bindingException = new SuperCsvBindingException("has binding error.", context, bindingErrors);
164             rowException.ifPresent(re -> bindingException.addAllProcessingErrors(re.getColumnErrors()));
165             
166             throw bindingException;
167             
168         }
169     }

171     /**
172      * Extracts the bean values, using the supplied name mapping array.
173      * 
174      * @param source
175      *            the bean
176      * @param nameMapping
177      *            the name mapping
178      * @throws NullPointerException
179      *             if source or nameMapping are null
180      * @throws SuperCsvReflectionException
181      *             if there was a reflection exception extracting the bean value
182      */
183     protected void extractBeanValues(final Object source, final String[] nameMapping) throws SuperCsvReflectionException {
184         
185         Objects.requireNonNull(nameMapping, "the nameMapping array can't be null as it's used to map from fields to columns");
186         
187         beanValues.clear();
188         
189         for( int i = 0; i < nameMapping.length; i++ ) {
190             
191             final String fieldName = nameMapping[i];
192             
193             if( fieldName == null ) {
194                 beanValues.add(null); // assume they always want a blank column
195                 
196             } else {
197                 Method getMethod = cache.getGetMethod(source, fieldName);
198                 try {
199                     beanValues.add(getMethod.invoke(source));
200                 }
201                 catch(final Exception e) {
202                     throw new SuperCsvReflectionException(String.format("error extracting bean value for field %s",
203                         fieldName), e);
204                 }
205             }
206             
207         }
208         
209     }

211     /**
212      * 行の各カラムの値に対して、CellProcessorを適用します。
213      * 
214      * @param destination
215      * @param source
216      * @param processors
217      * @param context
218      */
219     protected void executeCellProcessors(final List<Object> destination, final List<?> source,
220             final CellProcessor[] processors, final CsvContext context) {
221         
222         destination.clear();
223         
224         final SuperCsvRowException rowException = new SuperCsvRowException(
225                 String.format("row (%d) has errors column", context.getRowNumber()), context);
226         for( int i = 0; i < source.size(); i++ ) {
227             
228             try {
229                 context.setColumnNumber(i + 1); // update context (columns start at 1)
230                 
231                 if( processors[i] == null ) {
232                     destination.add(source.get(i)); // no processing required
233                 } else {
234                     destination.add(processors[i].execute(source.get(i), context)); // execute the processor chain
235                 }
236             } catch(SuperCsvCellProcessorException e) {
237                 rowException.addError(e);
238                 
239                 // 各カラムでエラーがあっても、後の入力値検証で処理を続けるために、仮に値を設定する。
240                 destination.add(source.get(i));
241                 
242             } catch(SuperCsvException e) {
243                 throw e;
244             }
245         }
246         
247         if(rowException.isNotEmptyColumnErrors()) {
248             throw rowException;
249         }
250         
251     }
