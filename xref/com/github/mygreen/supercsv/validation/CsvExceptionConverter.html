46      /**
47       * 例外をエラーオブジェクトに変換し、さらに、エラーオブジェジェクトをメッセージにフォーマットする。
48       * @param exception 変換するSuperCsvの例外。

50       * @return フォーマットされたメッセージ。
51       * @throws NullPointerException {@literal exception or beanMapping is null.}
52       */
53      public List<String> convertAndFormat(final SuperCsvException exception, final BeanMapping<?> beanMapping) {
54          
55          return convert(exception, beanMapping).stream()
56                  .map(error -> error.format(messageResolver, messageInterpolator))
57                  .collect(Collectors.toList());
58      }

60      /**
61       * 例外をエラーオブジェクトに変換する。
62       * @param exception 変換するSuperCsvの例外。
63       * @param beanMapping Beanの定義情報
64       * @return 変換されたエラーオブジェクト。
65       * @throws NullPointerException {@literal exception or beanMapping is null.}
66       */
67      public List<CsvError> convert(final SuperCsvException exception, final BeanMapping<?> beanMapping) {
68          
69          Objects.requireNonNull(beanMapping, "beanMapping should not be null.");
70          Objects.requireNonNull(exception, "exception should not be null.");
71          
72          final List<CsvError> errors = new ArrayList<>();
73          
74          if(exception instanceof SuperCsvBindingException) {
75              errors.addAll(convert((SuperCsvBindingException) exception, beanMapping));
76              
77          } if(exception instanceof SuperCsvRowException) {
78              errors.addAll(convert((SuperCsvRowException) exception, beanMapping));
79              
80          } else if(exception instanceof SuperCsvValidationException) {
81              errors.addAll(convert((SuperCsvValidationException)exception, beanMapping));
82              
83          } else if(exception instanceof SuperCsvCellProcessorException) {
84              errors.addAll(convert((SuperCsvCellProcessorException)exception, beanMapping));
85              
86          } else if(exception instanceof SuperCsvNoMatchColumnSizeException) {
87              errors.addAll(convert((SuperCsvNoMatchColumnSizeException)exception, beanMapping));
88              
89          } else if(exception instanceof SuperCsvNoMatchHeaderException) {
90              errors.addAll(convert((SuperCsvNoMatchHeaderException)exception, beanMapping));
91              
92          } else {
93              errors.addAll(convertDefault(exception, beanMapping));
94              
95          }
96          
97          return errors;
98      }

100     private List<CsvError> convert(final SuperCsvBindingException exception, final BeanMapping<?> beanMapping) {
101         
102         return exception.getBindingErrors().getAllErrors();
103     }

105     private List<CsvError> convert(final SuperCsvRowException exception, final BeanMapping<?> beanMapping) {
106         
107         return exception.getColumnErrors().stream()
108             .flatMap(e -> convert(e, beanMapping).stream())
109             .collect(Collectors.toList());
110         
111     }

113     private List<CsvFieldError> convert(final SuperCsvValidationException exception, final BeanMapping<?> beanMapping) {
114         
115         final CsvContext context = exception.getCsvContext();
116         final int columnNumber = context.getColumnNumber();
117         
118         final ColumnMapping columnMapping = beanMapping.getColumnMapping(columnNumber)
119                 .orElseThrow(() ->  new IllegalStateException(
120                         String.format("not found column definition with umber=%d.", columnNumber)));
121                 
122         final Map<String, Object> variables = new HashMap<>();
123         variables.put("lineNumber", context.getLineNumber());
124         variables.put("rowNumber", context.getRowNumber());
125         variables.put("columnNumber", context.getColumnNumber());
126         variables.put("label", columnMapping.getLabel());
127         variables.put("validatedValue", exception.getRejectedValue());
128         variables.putAll(exception.getMessageVariables());
129         
130         final String defaultMessage = exception.getValidationMessage();
131         final String errorCode = exception.getProcessor().getClass().getSimpleName();
132         final String objectName = beanMapping.getType().getSimpleName();
133         final String fieldName = columnMapping.getName();
134         
135         // 型変換エラーのコードを生成する
136         String[] typeMismatchCodes = {};
137         if(exception.isParedError()) {
138             // パース時の型変換エラーの場合
139             typeMismatchCodes = codeGenerator.generateTypeMismatchCodes(
140                     objectName, columnMapping.getName(), columnMapping.getField().getType());
141             
142         }
143         
144         // Bean名でエラーコードを生成する
145         String[] errorCodes = codeGenerator.generateCodes(
146                 errorCode, objectName, columnMapping.getName(), columnMapping.getField().getType());
147         
148         errorCodes = Utils.concat(typeMismatchCodes, errorCodes);
149         
150         final CsvFieldError fieldError = new CsvFieldError.Builder(objectName, fieldName, errorCodes)
151                 .processingFailure(true)
152                 .variables(variables)
153                 .defaultMessage(defaultMessage)
154                 .build();
155         
156         return Arrays.asList(fieldError);
157     }

159     private List<CsvFieldError> convert(final SuperCsvCellProcessorException exception, final BeanMapping<?> beanMapping) {
160         
161         final CsvContext context = exception.getCsvContext();
162         final int columnNumber = context.getColumnNumber();
163         final Object rejectedValue = context.getRowSource().get(columnNumber-1);
164         
165         final ColumnMapping columnMapping = beanMapping.getColumnMapping(columnNumber)
166                 .orElseThrow(() ->  new IllegalStateException(
167                         String.format("not found column definition with number=%d.", columnNumber)));
168                 
169         final Map<String, Object> variables = new HashMap<>();
170         variables.put("lineNumber", context.getLineNumber());
171         variables.put("rowNumber", context.getRowNumber());
172         variables.put("columnNumber", context.getColumnNumber());
173         variables.put("label", columnMapping.getLabel());
174         variables.put("validatedValue", rejectedValue);
175         
176         final String defaultMessage = exception.getMessage();
177         
178         final String errorCode = exception.getProcessor().getClass().getSimpleName();
179         final String objectName = beanMapping.getType().getSimpleName();
180         final String fieldName = columnMapping.getName();
181         
182         String[] errorCodes = codeGenerator.generateCodes(
183                 errorCode, objectName, columnMapping.getName(), columnMapping.getField().getType());
184         
185         final CsvFieldError fieldError = new CsvFieldError.Builder(objectName, fieldName, errorCodes)
186                 .processingFailure(true)
187                 .variables(variables)
188                 .defaultMessage(defaultMessage)
189                 .build();
190         
191         return Arrays.asList(fieldError);
