12  /**
13   * The resolves messages based on the registered {@link Properties}.
14   *
15   * @author T.TSUCHIE
16   *
17   */
18  public class PropertiesMessageResolver implements MessageResolver {
19      
20      protected Properties properties = new Properties();
21      
22      public PropertiesMessageResolver() {
23          this.properties = new Properties();
24      }
25      
26      public PropertiesMessageResolver(final Properties properties) {
27          
28          if(properties == null) {
29              throw new NullPointerException("properties should not be null.");
30          }
31          
32          this.properties = properties;
33      }
34      
35      /**
36       * {@inheritDoc}
37       */
38      public String getMessage(final String code) {
39          return properties.getProperty(code);
40      }
41  
42      
43      public Properties getProperties() {
44          return properties;
45      }
46      
47      public void setProperties(Properties properties) {
48          this.properties = properties;
49      }
