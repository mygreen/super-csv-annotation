22  public class ParseByte extends CellProcessorAdaptor implements StringCellProcessor {
23      
24      public ParseByte() {
25          super();
26      }
27      
28      public ParseByte(final LongCellProcessor next) {
29          super(next);
30      }
31      
32      @SuppressWarnings("unchecked")
33      @Override
34      public Object execute(final Object value, final CsvContext context) {
35          
36          validateInputNotNull(value, context);
37          
38          final Byte result;
39          if( value instanceof Byte ) {
40              result = (Byte) value;
41              
42          } else if( value instanceof String ) {
43              try {
44                  result = Byte.valueOf((String) value);
45              } catch(final NumberFormatException e) {
46                  throw new SuperCsvCellProcessorException(String.format("'%s' could not be parsed as an Byte", value),
47                      context, this, e);
48              }
49          } else {
50              final String actualClassName = value.getClass().getName();
51              throw new SuperCsvCellProcessorException(String.format(
52                  "the input value should be of type Byte or String but is of type %s", actualClassName), context, this);
53          }
54          
55          return next.execute(result, context);
56      }
57  }
