121     /**
122      * 組み立て途中の{@link CellProcessor}に最小値/最大値/範囲をチェックするための{@link CellProcessor}を、Chainの前に追加する。
123      * 
124      * @param type フィールドのクラスタイプ。
125      * @param annos フィールドに付与された全てのアノテーション。
126      * @param cellProcessor 組み立て途中の{@link CellProcessor}
127      * @param min 最小値
128      * @param max 最大値
129      * @return 最小値、最大値の指定の仕方により、最小値、最大値、範囲のチェックを追加するかどうか変わる。
130      */
131     protected CellProcessor prependRangeProcessor(final Class<N> type, final Annotation[] annos, final CellProcessor cellProcessor,
132             final Optional<N> min, final Optional<N> max) {
133         
134         final Optional<NumberFormat> formatter = createNumberFormatter(getNumberConverterAnnotation(annos));
135         
136         if(min.isPresent() && max.isPresent()) {
137             
138             final Range<N> cp;
139             if(cellProcessor == null) {
140                 cp = new Range<N>(min.get(), max.get());
141             } else {
142                 cp = new Range<N>(min.get(), max.get(), cellProcessor);
143             }
144             
145             formatter.ifPresent(f -> cp.setFormatter(f));
146             return cp;
147             
148         } else if(min.isPresent()) {
149             
150             final Min<N> cp;
151             if(cellProcessor == null) {
152                 cp = new Min<N>(min.get());
153             } else {
154                 cp = new Min<N>(min.get(), cellProcessor);
155             }
156             
157             formatter.ifPresent(f -> cp.setFormatter(f));
158             return cp;
159             
160         } else if(max.isPresent()) {
161             
162             final Max<N> cp;
163             if(cellProcessor == null) {
164                 cp = new Max<N>(max.get());
165             } else {
166                 cp = new Max<N>(max.get(), cellProcessor);
167             }
168             
169             formatter.ifPresent(f -> cp.setFormatter(f));
170             return cp;
171         }
172         
173         return cellProcessor;
174     }

176     /**
177      * 変換規則から、{@link NumberFormat}のインスタンスを作成する。
178      * @param converterAnno 変換規則を定義したアノテーション。
179      * @return 数値のフォーマッタ。アノテーションがなかったり、書式(pattern)が指定されていない場合は、空を返す。
180      */
181     protected Optional<NumberFormat> createNumberFormatter(final Optional<CsvNumberConverter> converterAnno) {
182         
183         final Optional<String> pattern = getPattern(converterAnno);
184         
185         if(!pattern.isPresent()) {
186             return Optional.empty();
187         }
188         
189 //        final boolean lenient = getLenient(converterAnno);
190         final Locale locale = getLocale(converterAnno);
191         final Optional<Currency> currency = getCurrency(converterAnno);
192         final Optional<RoundingMode> roundingMode = getRounding(converterAnno);
193         final DecimalFormatSymbols symbols = DecimalFormatSymbols.getInstance(locale);
194         
195         final DecimalFormat formatter = new DecimalFormat(pattern.get(), symbols);
196         formatter.setParseBigDecimal(true);
197         
198         roundingMode.ifPresent(r -> formatter.setRoundingMode(r));
199         currency.ifPresent(c -> formatter.setCurrency(c));
200         
201         return Optional.of(formatter);
202     }
